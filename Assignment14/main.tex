\section{OPTIMIZATION 2.2}
 Find the maximum profit that a company can
make, if the profit function is given by
$p(x)=41-72x-18x^2$

\section{SOLUTION}
\begin{lemma}
A function f(x) is said to be convex if following inequality is true for $\lambda \in [0,1] :$  \label{lemma1}
\begin{align}
    \lambda f(x_1) + (1-\lambda)f(x_2) \geq f(\lambda x_1 + (1-\lambda)x_2)
\end{align}
\end{lemma}
Given the profit function of the company is
\begin{equation}
    p(x)=41-72x-18x^2
\end{equation}
Checking convexity of $p(x)$ :
\begin{equation}
\begin{aligned}
    &\lambda\brak{41-72{x_1}-18{x_1}^2} + 
    (1-\lambda)\brak{41-72 {x_2}-18 {x_2}^2} \\ &\geq 
    \brak{41-72(\lambda x_1 + (1-\lambda)x_2)-18\brak{\lambda x_1 + (1-\lambda)x_2}^2} 
\end{aligned}
\end{equation}
resulting in
\begin{align}
    18\lambda(\lambda - 1)({x_1} -{x_2})^2 &\geq 0\\
    \implies \lambda(\lambda - 1)&\geq 0
\end{align}is not true .

$\implies$
The function is not convex.

Consider the derivative of $p(x)$,
\begin{equation}
    \frac{dp(x)}{dx}= -36x-72
\end{equation}
 Critical point is given by
\begin{align}
    \frac{dp(x)}{dx} &= 0 \\
    \implies x &= -2
\end{align}
is the point where the profit function will be maximum.
And the maximum profit will be $p(-2)$
\begin{equation}
    p(-2)=41-72(-2)-18(-2)^2=113
\end{equation}
\begin{align}
   \boxed{\text{Maxima} = 113}\\
   \boxed{\text{Maxima Point} = -2}
\end{align}
